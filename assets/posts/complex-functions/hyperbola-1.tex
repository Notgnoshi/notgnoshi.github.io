\documentclass[varwidth=40cm, border=5pt]{standalone}
\usepackage{tikz, pgfplots, amsmath, amssymb}
\pgfplotsset{compat=1.13}

\begin{document}
\begin{tikzpicture}[scale=2]
    \begin{axis}[
        axis equal,
        axis x line = middle,
        axis y line = middle,
        ticks = none,
        xlabel = {$x$},
        ylabel = {$y$},
        view={0}{90}
    ]
        \addplot3[very thick, domain=-2:2, contour gnuplot={labels=false, levels={1}}, contour/draw color={black}] {x^2 - y^2};
        \draw (0.7, 0.5) node[right, rotate=55]{$\displaystyle \frac{x^2}{c} - \frac{y^2}{c} = 1$};
    \end{axis}
    \begin{scope}[xshift=8cm]
        \begin{axis}[
            axis equal,
            axis x line = middle,
            axis y line = middle,
            ticks = none,
            xlabel = {$u$},
            ylabel = {$v$},
        ]
            \addplot[color=black, very thick] coordinates {(1, -2) (1, 2)};
            \draw (1, 1) node[right]{$u = c$};
        \end{axis}
    \end{scope}
\end{tikzpicture}
\end{document}
