\documentclass[tikz]{standalone}

\newcommand{\tikzAngleOfLine}{\tikz@AngleOfLine}                               
    \def\tikz@AngleOfLine(#1)(#2)#3{%                                            
    \pgfmathanglebetweenpoints{%                                                 
        \pgfpointanchor{#1}{center}}{%                                             
        \pgfpointanchor{#2}{center}}                                               
    \pgfmathsetmacro{#3}{\pgfmathresult}%                                        
    }
    
\newcommand{\tikzMarkAngle}[3]{                                                
    \tikzAngleOfLine#1#2{\AngleStart}                                              
    \tikzAngleOfLine#1#3{\AngleEnd}                                                
    \draw #1+(\AngleStart:0.4cm) arc (\AngleStart:\AngleEnd:0.4cm);              
}

    \begin{tikzpicture}[scale=1]
        \coordinate (O) at (0, 0);
        \coordinate (z) at (3, 3);
        \coordinate (a) at (3, 0);
        %axes
        \draw [->, thick] (-1, 0) -- (5, 0);
        \draw [->, thick] (0, -1) -- (0, 5);
        
        \draw (O) -- (z) node[circle, fill, inner sep=1pt, label={[right]$z$}](z){};
        \draw (O) -- (z) node[midway, label={[above]$r$}](r){};
        \draw (a) -- (z);
        
        
        \draw (O) node[label={[xshift=-0.3cm, yshift=-0.7cm]$O$}](O){};
        \draw (O) node[label={[xshift=0.6cm, yshift=-0.2cm]$\phi$}](O){};
        \tikzMarkAngle{(O)}{(z)}{(a)}
        \draw (3, 1.5) node[label=right:$b$](a){};
        \draw (1.5, 0) node[label=below:$a$](b){};
                
    \end{tikzpicture}
